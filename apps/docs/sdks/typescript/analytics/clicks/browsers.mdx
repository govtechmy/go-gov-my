title: dub.analytics.clicks.browsers
sidebarTitle: Browsers

---

import AnalyticsArguments from "/snippets/analytics-arguments.mdx";

<Note>
  Analytics endpoints require a [Pro plan](https://d.to/pro) subscription or
  higher.
</Note>

<AnalyticsArguments />

## Response

<ResponseField name="browser" type="string" required>
  The name of the browser
</ResponseField>

<ResponseField name="clicks" type="number" required>
  The number of clicks from this browser
</ResponseField>

<RequestExample>

```typescript
await dub.analytics.clicks.browsers({
  key: "dummy",
  domain: "dub.sh",
  interval: "30d",
});
```

</RequestExample>

<ResponseExample>

```typescript
[
  { browser: "Chrome", clicks: 19 },
  { browser: "Safari", clicks: 12 },
];
```

</ResponseExample>
